function setupAddressAutocomplete() { var input = document.getElementById('address'); var list = document.getElementById('address-suggestions'); if (!input || !list) return; var debounce; input.addEventListener('input', function() { clearTimeout(debounce); var q = input.value.trim(); if (q.length < 3) { list.classList.remove('open'); return; } debounce = setTimeout(function() { var url = 'https://api.mapbox.com/geocoding/v5/mapbox.places/' + encodeURIComponent(q) + '.json?access_token=' + MAPBOX_TOKEN + '&country=US&types=address&autocomplete=true&limit=5'; fetch(url) .then(function(r) { return r.json(); }) .then(function(data) { list.innerHTML = ''; if (!data.features || data.features.length === 0) { list.classList.remove('open'); return; } data.features.forEach(function(f) { var li = document.createElement('li'); li.textContent = f.place_name; li.addEventListener('mousedown', function(e) { e.preventDefault(); var parts = f.place_name.split(','); input.value = parts[0].trim(); var rest = parts.slice(1).join(',').trim(); var addr2 = document.getElementById('address2'); if (addr2) addr2.value = rest; list.classList.remove('open'); list.innerHTML = ''; }); list.appendChild(li); }); list.classList.add('open'); }) .catch(function() { list.classList.remove('open'); }); }, 300); }); document.addEventListener('click', function(e) { if (!input.contains(e.target) && !list.contains(e.target)) { list.classList.remove('open'); } }); input.addEventListener('keydown', function(e) { var items = list.querySelectorAll('li'); var active = list.querySelector('li.active'); var idx = Array.prototype.indexOf.call(items, active); if (e.key === 'ArrowDown') { e.preventDefault(); if (active) active.classList.remove('active'); var next = items[idx + 1] || items[0]; if (next) next.classList.add('active'); } else if (e.key === 'ArrowUp') { e.preventDefault(); if (active) active.classList.remove('active'); var prev = items[idx - 1] || items[items.length - 1]; if (prev) prev.classList.add('active'); } else if (e.key === 'Enter' && active) { e.preventDefault(); active.dispatchEvent(new MouseEvent('mousedown')); } else if (e.key === 'Escape') { list.classList.remove('open'); } }); }
